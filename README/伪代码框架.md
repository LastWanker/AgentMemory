def build_chat_memory(config):
    turns = extract_user_turns(conversations_json, session_id=config.session_id)

    for t in turns:
        t.text, reason = clean_and_flag(t.text)
        if reason:
            t.is_excluded = True
            t.exclude_reason = reason

    for session_turns in group_by_session(turns):
        assign_local_clusters(
            session_turns,
            mode=config.segmentation_mode,          # adaptive / fixed
            fixed_sim_threshold=config.fixed_sim_threshold,
            novelty_threshold=config.novelty_threshold,
            min_segment_len=config.min_segment_len,
        )

    assign_global_clusters(
        turns,
        enable_cross_session=config.cross_session_merge,
        merge_similarity_threshold=config.global_merge_similarity_threshold,
    )

    raw_rows = [to_row(t, turns) for t in turns]
    dedup_turns = dedup_non_excluded(turns, dedup_min_chars=8)
    dedup_rows = [to_row(t, dedup_turns) for t in dedup_turns]
    memory_rows = to_memory_rows(dedup_turns)      # mem_id = c000001...

    write_jsonl(raw_out, raw_rows)
    write_jsonl(dedup_out, dedup_rows)
    write_jsonl(memory_out, memory_rows)
```

async def build_llm_supplemental_rows(memories, api_key, cfg):
    clusters = group_by_cluster_id(memories, min_cluster_size=cfg.min_cluster_size)
    clusters = clusters[:cfg.max_clusters] if cfg.max_clusters else clusters

    rows = []
    for cluster in clusters:
        k = ceil(len(cluster.mem_ids) / 3)
        k = min(k, cfg.max_per_layer) if cfg.max_per_layer else k

        for layer in ("object", "stance", "intent"):
            if cfg.fallback_only:
                query_texts, mode = fallback_queries(cluster, layer, k), "fallback"
            else:
                query_texts, mode = llm_with_retry_or_fallback(cluster, layer, k, cfg)

            rows += [
                {
                    "query_id": f"llmq-{cluster.cluster_id}-{layer}-{i:03d}",
                    "query_text": text,
                    "positives": cluster.mem_ids,
                    "meta": {
                        "source": "chat_llm_supplemental",
                        "channel": "supplemental",
                        "layer": layer,
                        "generation_mode": mode,
                    },
                }
                for i, text in enumerate(query_texts, start=1)
            ]
    return sorted(rows, key=lambda row: row["query_id"])
```

```python
def retrieve(query_text, candidate_mode):
    q = encode_query(query_text)  # q_vecs + coarse_vec + lex_tokens + coarse_role=query

    coarse_candidates = coarse_index.search(q.coarse_vec, top_n)
    lexical_candidates = lexical_index.search(q.lex_tokens, top_n) if lexical_index else []
    candidates = select_candidates(candidate_mode, coarse_candidates, lexical_candidates)

    scored = []
    for mem_id, coarse_score in candidates:
        semantic_score, debug = scorer.score_or_score_many(q.q_vecs, mem_vecs[mem_id])
        lexical_score = lexical_score_or_token_overlap(mem_id, q.lex_tokens)
        meta_score = avg(source_score, tag_match, recency_score)
        scored.append((mem_id, features_dict, debug))

    route_output = router.route(scored)            # soft / half_hard / hard
    results = apply_policy_weight(route_output, scored)
    return topk(results)

def train_reranker_main(args):
    if args.candidate_mode != "coarse":
        args.candidate_mode = "coarse"             # 当前固定口径

    items = load_memory_items(memory_path)
    eval_queries = load_eval_queries(eval_path)    # positives 或 expected_mem_ids
    eval_queries = eval_queries[:args.max_queries] if args.max_queries > 0 else eval_queries

    store, index, lexical_index = build_memory_index(..., cache_signature=train_cache_sig)
    retriever = Retriever(router=Router(policy="soft", top_k=args.top_n), scorer=FieldScorer())

    pair_buffer, listwise_buffer, cardinality_samples = [], [], []
    for row in eval_queries:
        q = encode_query(row["query_text"])
        results = retriever.retrieve(q, top_n=args.top_n, top_k=args.top_n, candidate_mode="coarse")
        candidates = dedup_mem_ids(results)

        # 支持 chunked mem_id：is_positive("c000001#c1", ["c000001"]) == True
        positives = [mid for mid in candidates if is_positive(mid, row["positives"])]
        negatives = build_negatives(candidates, positives, strategy=args.neg_strategy)

        if args.loss_type == "pairwise":
            pair_buffer += build_pair_samples(q, positives, negatives, neg_per_pos=args.neg_per_pos)
        else:
            listwise_buffer += build_listwise_samples(q, candidates, positives, list_size=args.list_size)
        cardinality_samples += build_cardinality_samples(q, positives, k_max=args.cardinality_k_max)

    train_tiny_reranker(model, pair_buffer_or_listwise_buffer, cardinality_samples)
    save_weight_payload(model_state, cardinality_state, meta=args)
    write_train_metrics_json(run_dir)
    upsert_runs_registry_csv(run_dir)
```
